def Python_random_generator(Nu):
    x = np.random.normal(0, 1,100)
    for count,i in enumerate(x,1):
        if count <= Nu:
            print(i)
    print("")
    mean = sum(x[:Nu])/Nu
    print(f"Mean: {mean}")
    print("")
    median = np.median((x[:Nu]))
    print(f"Median: {median}")
    print("")      
    sd = np.std(x[:Nu])
    print(f"Standard deviation: {sd}")
